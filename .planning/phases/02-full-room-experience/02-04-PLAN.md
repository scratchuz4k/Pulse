      Import ref, onMounted, onUnmounted from 'vue'.

      Declare CODE_TO_ACCELERATOR map with entries:
        Space: 'Space', Enter: 'Return', Backspace: 'Backspace', Tab: 'Tab', Escape: 'Escape',
        ArrowUp: 'Up', ArrowDown: 'Down', ArrowLeft: 'Left', ArrowRight: 'Right'

      Export function codeToAccelerator(code: string): string that:
        checks the map, then strips 'Key' from 'Key[A-Z]', strips 'Digit' from 'Digit[0-9]', passes F-keys through, else returns code as-is.

      Export interface PttBinding { accelerator: string; label: string }

      Export function usePtt():
        const isPttMode = ref(false)
        const pttBinding = ref&lt;PttBinding | null&gt;(null)
        const isCapturing = ref(false)

        onMounted(async () => {
          // Load saved key
          const saved = await window.pulseApi.getPttKey()
          if (saved) pttBinding.value = { accelerator: saved, label: saved }
          // Register keyup listener for when window IS focused
          window.addEventListener('keyup', handleWindowKeyUp)
        })

        onUnmounted(() => {
          window.pulseApi.removePttListeners()
          window.removeEventListener('keyup', handleWindowKeyUp)
        })

        function handleWindowKeyUp(e: KeyboardEvent): void {
          if (!isPttMode.value || !pttBinding.value) return
          const acc = codeToAccelerator(e.code)
          if (acc === pttBinding.value.accelerator) {
            releasePtt()
          }
        }

        function startCapture(): void { isCapturing.value = true }

        function handleCaptureKeydown(e: KeyboardEvent): void {
          if (!isCapturing.value) return
          e.preventDefault(); e.stopPropagation()
          const accelerator = codeToAccelerator(e.code)
          const label = e.code === 'Space' ? 'Space' : (e.key.length === 1 ? e.key.toUpperCase() : e.key)
          pttBinding.value = { accelerator, label }
          isCapturing.value = false
          window.pulseApi.setPttKey(accelerator)
        }

        // These are called by the component when IPC ptt events fire:
        function activatePtt(setMicFn: (v: boolean) => Promise&lt;void&gt;): void {
          setMicFn(true)
        }
        function releasePtt(setMicFn?: (v: boolean) => Promise&lt;void&gt;): void {
          setMicFn?.(false)
        }

        // Simpler: expose raw refs and let the component wire IPC to setMicrophoneEnabled directly.
        // The component calls window.pulseApi.onPttKeyDown(() => ...) and window.pulseApi.onPttKeyUp(() => ...) with the toggleMic bridge.

        return { isPttMode, pttBinding, isCapturing, startCapture, handleCaptureKeydown }

| globalShortcut → OS | OS-level key registration; malformed accelerator string is rejected by Electron |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02D-01 | Tampering | Accelerator string via ptt:set-key IPC | mitigate | In ipcMain handler: validate accelerator is a non-empty string before calling globalShortcut.register; globalShortcut.register returns false for invalid strings — check return value |
| T-02D-02 | Denial of Service | globalShortcut registration blocking OS keys | accept | User explicitly chooses the key; Electron registers it; this is expected behavior |
| T-02D-03| Tampering | IPC listener accumulation (Pitfall 3 from research) | mitigate | removePttListeners() called in onUnmounted and before re-adding in onMounted |
</threat_model>
